36int libwifi_create_deauth(struct libwifi_deauth *deauth,
37 const unsigned char receiver[6],
38 const unsigned char transmitter[6],
39 const unsigned char address3[6],
40 uint16_t reason_code) {
41 memset(deauth, 0, sizeof(struct libwifi_deauth));
42
43 deauth->frame_header.frame_control.type = TYPE_MANAGEMENT;
44 deauth->frame_header.frame_control.subtype = SUBTYPE_DEAUTH;
45 memcpy(&deauth->frame_header.addr1, receiver, 6);
46 memcpy(&deauth->frame_header.addr2, transmitter, 6);
47 memcpy(&deauth->frame_header.addr3, address3, 6);
48
49 deauth->frame_header.seq_control.sequence_number = (rand() % 4096);
50
51 memcpy(&deauth->fixed_parameters.reason_code, &reason_code, sizeof(reason_code));
52
53 return 0;
54}

60size_t libwifi_dump_deauth(struct libwifi_deauth *deauth, unsigned char *buf, size_t buf_len) {
61 size_t deauth_len = libwifi_get_deauth_length(deauth);
62 if (deauth_len > buf_len) {
63 return -EINVAL;
64 }
65
66 size_t offset = 0;
67 memcpy(buf + offset, &deauth->frame_header, sizeof(struct libwifi_mgmt_unordered_frame_header));
68 offset += sizeof(struct libwifi_mgmt_unordered_frame_header);
69
70 memcpy(buf + offset, &deauth->fixed_parameters, sizeof(struct libwifi_deauth_fixed_parameters));
71 offset += sizeof(struct libwifi_deauth_fixed_parameters);
72
73 memcpy(buf + offset, deauth->tags.parameters, deauth->tags.length);
74 offset += deauth->tags.length;
75
76 return deauth_len;
77}

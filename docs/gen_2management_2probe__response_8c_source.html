42int libwifi_set_probe_resp_ssid(struct libwifi_probe_resp *probe_resp, const char *ssid) {
43 int ret = 0;
44
45 if (probe_resp->tags.length != 0) {
46 ret = libwifi_remove_tag(&probe_resp->tags, TAG_SSID);
47 if (ret != 0) {
48 return ret;
49 }
50 }
51
52 ret = libwifi_quick_add_tag(&probe_resp->tags, TAG_SSID, (const unsigned char *) ssid, strlen(ssid));
53
54 return ret;
55}

81int libwifi_create_probe_resp(struct libwifi_probe_resp *probe_resp,
82 const unsigned char receiver[6],
83 const unsigned char transmitter[6],
84 const unsigned char address3[6],
85 const char *ssid,
86 uint8_t channel) {
87 memset(probe_resp, 0, sizeof(struct libwifi_probe_resp));
88
89 probe_resp->frame_header.frame_control.type = TYPE_MANAGEMENT;
90 probe_resp->frame_header.frame_control.subtype = SUBTYPE_PROBE_RESP;
91 memcpy(&probe_resp->frame_header.addr1, receiver, 6);
92 memcpy(&probe_resp->frame_header.addr2, transmitter, 6);
93 memcpy(&probe_resp->frame_header.addr3, address3, 6);
94
95 probe_resp->frame_header.seq_control.sequence_number = (rand() % 4096);
96 probe_resp->fixed_parameters.timestamp = BYTESWAP64(libwifi_get_epoch());
97 uint16_t probe_resp_interval = 50 + (rand() % 100);
98 probe_resp->fixed_parameters.probe_resp_interval = BYTESWAP16(probe_resp_interval);
99 probe_resp->fixed_parameters.capabilities_information = BYTESWAP16(LIBWIFI_DEFAULT_AP_CAPABS);
100
101 int ret = libwifi_set_probe_resp_ssid(probe_resp, ssid);
102 if (ret != 0) {
103 return ret;
104 }
105
106 ret = libwifi_set_probe_resp_channel(probe_resp, channel);
107
108 return ret;
109}

115size_t libwifi_dump_probe_resp(struct libwifi_probe_resp *probe_resp, unsigned char *buf, size_t buf_len) {
116 size_t probe_resp_len = libwifi_get_probe_resp_length(probe_resp);
117 if (probe_resp_len > buf_len) {
118 return -EINVAL;
119 }
120
121 size_t offset = 0;
122 memcpy(buf + offset, &probe_resp->frame_header, sizeof(struct libwifi_mgmt_unordered_frame_header));
123 offset += sizeof(struct libwifi_mgmt_unordered_frame_header);
124
125 memcpy(buf + offset, &probe_resp->fixed_parameters, sizeof(struct libwifi_probe_resp_fixed_parameters));
126 offset += sizeof(struct libwifi_probe_resp_fixed_parameters);
127
128 memcpy(buf + offset, probe_resp->tags.parameters, probe_resp->tags.length);
129 offset += probe_resp->tags.length;
130
131 return probe_resp_len;
132}
